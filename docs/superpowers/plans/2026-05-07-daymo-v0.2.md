# Daymo v0.2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Split Daymo's pipeline into a slow `capture` stage and a fast `compose` stage, then add scene transitions, intro/outro slates, dead-time primitives (`fx.fastForward`, `fx.skip`), and an opt-in per-scene capture mode.

**Architecture:** `daymo capture` records a self-describing artifact bundle (page video + events + `capture.json` manifest). `daymo compose` consumes the bundle plus the current `.demo` file and produces `output.mp4`. `daymo render` chains them. The compositor builds an ffmpeg filter graph that splits scene segments, applies per-scene transitions, prepends/appends slates rendered via a small headless Chromium pass, and applies fastForward/skip markers via `setpts`/`trim`.

**Tech Stack:** Same as v0.1 — Node.js ≥20.10 · TypeScript (ESM, NodeNext) · Playwright · ffmpeg · vitest · gray-matter · cac · execa · yaml.

**Spec:** `docs/superpowers/specs/2026-05-07-daymo-v0.2-compose-stage-design.md`

**Boundary rules (carried over + new):**
- `parser.ts` stays pure (string in, AST out — no I/O).
- `controller.ts` is the only Playwright consumer.
- `slates.ts` is the only **other** Playwright consumer (a tiny boot for slate rendering); shares no state with `controller.ts`.
- `transitions.ts` is pure (transition config in, ffmpeg filter fragment + duration math out).
- `manifest.ts` owns the `capture.json` schema and is the only module that reads/writes it.
- `compositor.ts` orchestrates ffmpeg only; it imports from `transitions.ts` and `slates.ts` but doesn't know about the `.demo` AST directly — it consumes the manifest plus a small `ComposePlan` struct the runner builds from the AST + manifest.

```ts
export function parseDurationMs(s: string | undefined, defaultMs: number): number {
  if (!s) return defaultMs;
  const m = /^([0-9.]+)\s*(ms|s)?$/.exec(s.trim());
  if (!m) throw new Error(`invalid duration "${s}"`);
  const n = Number(m[1]);
  return m[2] === "ms" ? n : n * 1000;
}
```

const DEFAULT_SLATE_BG = "#0a0a0a";
const DEFAULT_SLATE_ACCENT = "#3b82f6";

function normalizeSlate(
  raw: unknown,
  defaultDurationMs: number,
): SlateConfig | false | undefined {
  if (raw === undefined) return undefined;
  if (raw === false) return false;
  if (typeof raw !== "object" || raw === null) {
    throw new Error(`intro/outro must be an object or false`);
  }
  const r = raw as Record<string, unknown>;
  return {
    durationMs: parseDurationMs(typeof r.duration === "string" ? r.duration : undefined, defaultDurationMs),
    background: typeof r.background === "string" ? r.background : DEFAULT_SLATE_BG,
    accent: typeof r.accent === "string" ? r.accent : DEFAULT_SLATE_ACCENT,
    logo: typeof r.logo === "string" ? r.logo : undefined,
    title: typeof r.title === "string" ? r.title : undefined,
    subtitle: typeof r.subtitle === "string" ? r.subtitle : undefined,
    text: typeof r.text === "string" ? r.text : undefined,
  };
}
```

const FF_MIN = 1.5;
const FF_MAX = 16;

let activeMarker: "fast_forward" | "skip" | null = null;

// ... inside createFx, in the returned object:

async fastForward<T>(fn: () => Promise<T>, factor = 3): Promise<T> {
  if (activeMarker) {
    throw new Error(`fx.fastForward cannot be nested inside fx.${activeMarker}`);
  }
  const clamped = Math.max(FF_MIN, Math.min(FF_MAX, factor));
  events.push({ kind: "fast_forward_start", t: clock(), sceneIndex: -1, factor: clamped });
  activeMarker = "fast_forward";
  try {
    return await fn();
  } finally {
    events.push({ kind: "fast_forward_end", t: clock(), sceneIndex: -1 });
    activeMarker = null;
  }
},
```

`sceneIndex: -1` is filled in by the controller before serialization (controller knows the active scene; fx doesn't). Document this with a comment.

Note: `activeMarker` must be **per-fx-instance**, not module-level (the controller may create one fx per scene). Move it to a closure variable inside `createFx`.

// src/manifest.ts
import path from "node:path";
import fs from "node:fs/promises";
import type { CaptureMode, RunnerEvent, Scene } from "./types.js";

export interface ManifestSceneEntry {
  index: number;          // 0-based, dense
  title: string;
  slug: string;
  sourceLine: number;     // matches Scene.sourceLine for cross-ref
  tStartMs: number;
  tEndMs: number;
}

export type ManifestMarker =
  | { kind: "fast_forward"; sceneIndex: number; tStartMs: number; tEndMs: number; factor: number }
  | { kind: "skip";         sceneIndex: number; tStartMs: number; tEndMs: number };

export interface Manifest {
  version: 2;
  demoFile: string;
  captureMode: CaptureMode;
  viewport: { width: number; height: number };
  createdAt: string;
  scenes: ManifestSceneEntry[];
  markers: ManifestMarker[];
}

export function slugify(title: string): string {
  return title.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/^-+|-+$/g, "");
}

export interface BuildManifestArgs {
  demoFile: string;
  captureMode: CaptureMode;
  viewport: { width: number; height: number };
  scenes: Scene[];
  events: RunnerEvent[];
}

export function buildManifest(args: BuildManifestArgs): Manifest {
  const sourceLineToIndex = new Map(args.scenes.map((s, i) => [s.sourceLine, i]));
  const sceneEntries: ManifestSceneEntry[] = args.scenes.map((s, i) => ({
    index: i,
    title: s.title,
    slug: slugify(s.title),
    sourceLine: s.sourceLine,
    tStartMs: 0,
    tEndMs: 0,
  }));
  for (const ev of args.events) {
    if (ev.kind === "scene_start") {
      const i = sourceLineToIndex.get(ev.index);
      if (i !== undefined) sceneEntries[i].tStartMs = ev.t;
    } else if (ev.kind === "scene_end") {
      const i = sourceLineToIndex.get(ev.index);
      if (i !== undefined) sceneEntries[i].tEndMs = ev.t;
    }
  }

  const markers: ManifestMarker[] = [];
  let pendingFf: { tStart: number; sceneIndex: number; factor: number } | null = null;
  let pendingSkip: { tStart: number; sceneIndex: number } | null = null;
  for (const ev of args.events) {
    if (ev.kind === "fast_forward_start") {
      pendingFf = { tStart: ev.t, sceneIndex: sourceLineToIndex.get(ev.sceneIndex) ?? 0, factor: ev.factor };
    } else if (ev.kind === "fast_forward_end" && pendingFf) {
      markers.push({ kind: "fast_forward", ...{ sceneIndex: pendingFf.sceneIndex, tStartMs: pendingFf.tStart, tEndMs: ev.t, factor: pendingFf.factor } });
      pendingFf = null;
    } else if (ev.kind === "skip_start") {
      pendingSkip = { tStart: ev.t, sceneIndex: sourceLineToIndex.get(ev.sceneIndex) ?? 0 };
    } else if (ev.kind === "skip_end" && pendingSkip) {
      markers.push({ kind: "skip", sceneIndex: pendingSkip.sceneIndex, tStartMs: pendingSkip.tStart, tEndMs: ev.t });
      pendingSkip = null;
    }
  }

  return {
    version: 2,
    demoFile: args.demoFile,
    captureMode: args.captureMode,
    viewport: args.viewport,
    createdAt: new Date().toISOString(),
    scenes: sceneEntries,
    markers,
  };
}

export async function writeManifest(captureDir: string, m: Manifest): Promise<void> {
  await fs.writeFile(path.join(captureDir, "capture.json"), JSON.stringify(m, null, 2));
}

export async function readManifest(captureDir: string): Promise<Manifest> {
  const txt = await fs.readFile(path.join(captureDir, "capture.json"), "utf8");
  const m = JSON.parse(txt) as Manifest;
  if (m.version !== 2) throw new Error(`unsupported manifest version: ${m.version}`);
  return m;
}
```

import path from "node:path";
import fs from "node:fs/promises";
import crypto from "node:crypto";
import { parse } from "./parser.js";
import { Controller } from "./controller.js";
import { compose } from "./compositor.js";
import { buildManifest, writeManifest, readManifest } from "./manifest.js";
import type { ArtifactPaths, CaptureBundle, DemoAst } from "./types.js";

export interface CaptureOpts {
  demoFile: string;
  artifactsBase?: string;           // default ./artifacts
  onlyScene?: number;               // 0-based, per-scene mode only
}

export interface ComposeOpts {
  bundleDir: string;                // an artifacts/<id> dir
  demoFileOverride?: string;        // optional .demo override; defaults to manifest.demoFile
}

function buildArtifactPaths(rootDir: string): ArtifactPaths {
  const captureDir = path.join(rootDir, "capture");
  return {
    dir: rootDir,
    capture: {
      dir: captureDir,
      manifest: path.join(captureDir, "capture.json"),
      events: path.join(captureDir, "events.json"),
      rawVideo: path.join(captureDir, "page.mp4"),
      scenesDir: path.join(captureDir, "scenes"),
    },
    output: path.join(rootDir, "output.mp4"),
    composeLog: path.join(rootDir, "compose.log"),
  };
}

export async function capture(opts: CaptureOpts): Promise<{ artifactsDir: string; ast: DemoAst }> {
  const source = await fs.readFile(opts.demoFile, "utf8");
  const ast = parse(source);
  const id = crypto.randomBytes(4).toString("hex");
  const artifactsDir = path.resolve(opts.artifactsBase ?? "./artifacts", id);
  const paths = buildArtifactPaths(artifactsDir);
  await fs.mkdir(paths.capture.dir, { recursive: true });

  const ctrl = await Controller.start({
    url: ast.frontmatter.url,
    viewport: ast.frontmatter.viewport,
    mocks: ast.frontmatter.mocks,
    storageStatePath: ast.frontmatter.auth?.storageState
      ? path.resolve(path.dirname(opts.demoFile), ast.frontmatter.auth.storageState)
      : undefined,
    artifactsDir: paths.capture.dir,
  });
  try {
    for (const scene of ast.scenes) await ctrl.runScene(scene);
  } finally {
    await ctrl.stop();
  }

  const events = JSON.parse(await fs.readFile(paths.capture.events, "utf8"));
  const m = buildManifest({
    demoFile: path.resolve(opts.demoFile),
    captureMode: ast.frontmatter.captureMode ?? "continuous",
    viewport: ast.frontmatter.viewport ?? { width: 1440, height: 900 },
    scenes: ast.scenes,
    events,
  });
  await writeManifest(paths.capture.dir, m);
  return { artifactsDir, ast };
}

export async function composeFromBundle(opts: ComposeOpts): Promise<{ mp4Path: string }> {
  const paths = buildArtifactPaths(opts.bundleDir);
  const m = await readManifest(paths.capture.dir);
  const demoFile = opts.demoFileOverride ?? m.demoFile;
  const source = await fs.readFile(demoFile, "utf8");
  const ast = parse(source);
  const baseDir = path.dirname(demoFile);
  const musicSrc = ast.frontmatter.music ? path.resolve(baseDir, ast.frontmatter.music) : null;
  await compose({ paths, manifest: m, ast, baseDir, musicSrc });
  return { mp4Path: paths.output };
}

export async function render(opts: CaptureOpts): Promise<{ mp4Path: string; artifactsDir: string }> {
  const { artifactsDir } = await capture(opts);
  const { mp4Path } = await composeFromBundle({ bundleDir: artifactsDir });
  return { mp4Path, artifactsDir };
}

  manifest: Manifest;

  // Phase 4–6 will replace this with a full filter graph builder.

      "-i", opts.musicSrc,

      "-map", "[m]",
      "-c:v", "libx264",
      "-c:a", "aac",

      opts.paths.output,

// src/transitions.ts
import type { TransitionConfig, TransitionType } from "./types.js";

const XFADE_NAME: Record<Exclude<TransitionType, "none">, string> = {
  "crossfade":   "fade",
  "dip-to-black": "fadeblack",
  "slide-left":  "slideleft",
  "slide-right": "slideright",
};

export interface BuildTransitionArgs {
  inLabelA: string;        // e.g. "[v0]"
  inLabelB: string;        // e.g. "[v1]"
  clipADurationMs: number;
  transition: TransitionConfig;
  outLabel: string;        // e.g. "[v01]"
}

export interface BuildTransitionResult {
  filter: string;
  outputDurationMs: number; // duration of the joined clip relative to clip A's timeline
}

export function buildTransitionFilter(args: BuildTransitionArgs): BuildTransitionResult {
  const { inLabelA, inLabelB, clipADurationMs, transition, outLabel } = args;
  if (transition.type === "none") {
    return {
      filter: `${inLabelA}${inLabelB}concat=n=2:v=1:a=0${outLabel}`,
      outputDurationMs: clipADurationMs,
    };
  }
  const name = XFADE_NAME[transition.type];
  const durationS = (transition.durationMs / 1000).toFixed(3);
  const offsetS = ((clipADurationMs - transition.durationMs) / 1000).toFixed(3);
  return {
    filter: `${inLabelA}${inLabelB}xfade=transition=${name}:duration=${durationS}:offset=${offsetS}${outLabel}`,
    outputDurationMs: clipADurationMs + clipADurationMs - transition.durationMs,
  };
}
```

The `outputDurationMs` formula is correct only when both clips have the same length; in practice the compositor will know each clip's duration and recompute joined durations as it stacks transitions. The unit test reflects clip-A duration only for `none`.

import { buildTransitionFilter } from "./transitions.js";
import { parseDurationMs } from "./parser.js";
import type { TransitionConfig } from "./types.js";

function resolveTransition(ast: DemoAst, sceneIndex: number): TransitionConfig {
  const scene = ast.scenes[sceneIndex] as any;
  if (scene.transition) return scene.transition;
  const type = (ast.frontmatter as any).defaultTransition ?? "crossfade";
  const durationMs = parseDurationMs((ast.frontmatter as any).transitionDuration, 500);
  return { type, durationMs };
}

interface ClipSpec {
  inputIndex: number;     // index into ffmpeg -i list
  trimStartMs?: number;
  trimEndMs?: number;
  durationMs: number;
}

function buildSceneClips(m: Manifest): ClipSpec[] {
  // continuous mode only for now; per-scene mode handled in Phase 7.
  return m.scenes.map(s => ({
    inputIndex: 0,                    // single -i page.webm
    trimStartMs: s.tStartMs,
    trimEndMs: s.tEndMs,
    durationMs: s.tEndMs - s.tStartMs,
  }));
}

function buildFilterGraph(opts: BuildArgsOpts): { filter: string; vOut: string; durationMs: number } {
  const clips = buildSceneClips(opts.manifest);
  const segments: string[] = [];
  // 1. Trim each scene out of the single input.
  clips.forEach((c, i) => {
    const startS = ((c.trimStartMs ?? 0) / 1000).toFixed(3);
    const endS   = ((c.trimEndMs   ?? 0) / 1000).toFixed(3);
    segments.push(`[0:v]trim=start=${startS}:end=${endS},setpts=PTS-STARTPTS[s${i}]`);
  });
  // 2. Join with transitions.
  let prevLabel = `[s0]`;
  let prevDuration = clips[0].durationMs;
  for (let i = 1; i < clips.length; i++) {
    const t = resolveTransition(opts.ast, i);
    const out = `[v${i}]`;
    const r = buildTransitionFilter({
      inLabelA: prevLabel,
      inLabelB: `[s${i}]`,
      clipADurationMs: prevDuration,
      transition: t,
      outLabel: out,
    });
    segments.push(r.filter);
    prevLabel = out;
    prevDuration = prevDuration + clips[i].durationMs - (t.type === "none" ? 0 : t.durationMs);
  }
  return { filter: segments.join(";"), vOut: prevLabel, durationMs: prevDuration };
}

export function buildFfmpegArgs(opts: BuildArgsOpts): string[] {
  const argv: string[] = ["-y", "-fflags", "+bitexact", "-map_metadata", "-1"];
  argv.push("-i", opts.paths.capture.rawVideo);
  if (opts.musicSrc) argv.push("-i", opts.musicSrc);

  const g = buildFilterGraph(opts);
  let filter = g.filter;
  let vMap = g.vOut;

  if (opts.musicSrc) {
    const vol = (opts.musicVolume ?? 0.4).toFixed(1);
    filter += `;[1:a]volume=${vol},atrim=end=${(g.durationMs / 1000).toFixed(3)}[m]`;
  }

  argv.push("-filter_complex", filter);
  argv.push("-map", vMap);
  if (opts.musicSrc) argv.push("-map", "[m]", "-c:a", "aac", "-shortest");
  else argv.push("-an");
  argv.push("-c:v", "libx264", opts.paths.output);
  return argv;
}

export interface ResolvedSlate extends SlateConfig {
  kind: "intro" | "outro";
}

export function resolveIntroConfig(fm: Frontmatter, raw: SlateInput): ResolvedSlate | null {
  if (raw === false) return null;
  const base: SlateConfig = {
    durationMs: 2500,
    background: "#0a0a0a",
    accent: "#3b82f6",
    title: fm.title,
    subtitle: fm.description,
  };
  const merged = raw === undefined ? base : { ...base, ...raw,
    title: raw.title ?? fm.title,
    subtitle: raw.subtitle ?? fm.description,
  };
  return { kind: "intro", ...merged };
}

export function resolveOutroConfig(fm: Frontmatter, raw: SlateInput): ResolvedSlate | null {
  if (raw === false) return null;

    durationMs: 2000,
    background: "#0a0a0a",
    accent: "#3b82f6",
    title: fm.title,
    text: "Made with Daymo",
  };
  const merged = raw === undefined ? base : { ...base, ...raw, title: raw.title ?? fm.title };
  return { kind: "outro", ...merged };
}

export function buildSlateHtml(s: ResolvedSlate): string {

import { chromium } from "playwright";
import { execa } from "execa";
import path from "node:path";
import fs from "node:fs/promises";

export interface RenderSlateArgs {
  slate: ResolvedSlate;
  viewport: { width: number; height: number };
  outDir: string;        // capture dir
  filename: string;      // e.g. "intro.mp4"
}

export async function renderSlate(args: RenderSlateArgs): Promise<string> {
  const html = buildSlateHtml(args.slate);
  const htmlPath = path.join(args.outDir, args.filename.replace(/\.mp4$/, ".html"));
  const webmPath = path.join(args.outDir, args.filename.replace(/\.mp4$/, ".webm"));
  const mp4Path  = path.join(args.outDir, args.filename);
  await fs.writeFile(htmlPath, html);

  const browser = await chromium.launch({ headless: true });
  try {
    const ctx = await browser.newContext({
      viewport: args.viewport,
      recordVideo: { dir: args.outDir, size: args.viewport },
    });
    const page = await ctx.newPage();
    await page.goto(`file://${htmlPath}`);
    await page.waitForTimeout(args.slate.durationMs);
    // page.video().path() only resolves AFTER context close.
    const video = page.video();
    await ctx.close();
    const recordedWebm = video ? await video.path() : "";
    if (recordedWebm) await fs.rename(recordedWebm, webmPath);
    // Convert webm → mp4 with the right duration trim.
    await execa("ffmpeg", [
      "-y", "-i", webmPath,
      "-t", (args.slate.durationMs / 1000).toFixed(3),
      "-c:v", "libx264", "-an",
      mp4Path,
    ]);
    await fs.unlink(webmPath).catch(() => {});
  } finally {
    await browser.close();
  }
  return mp4Path;
}

function buildFilterGraph(opts: BuildArgsOpts): { filter: string; vOut: string; durationMs: number; numInputs: number } {
  const sceneClips = buildSceneClips(opts.manifest);
  const introResolved = resolveIntroConfig(opts.ast.frontmatter, (opts.ast.frontmatter as any).intro);
  const outroResolved = resolveOutroConfig(opts.ast.frontmatter, (opts.ast.frontmatter as any).outro);
  const includeIntro = introResolved !== null && opts.slatePaths?.intro;
  const includeOutro = outroResolved !== null && opts.slatePaths?.outro;

  const segments: string[] = [];
  // Inputs are: [0]=page.webm, [1]=music?, [N]=intro?, [N+1]=outro?
  // Actually split: video inputs come first, music last. Use a small index map.
  let nextInput = 1;                       // 0 is page.webm
  if (opts.musicSrc) nextInput++;          // music takes one input
  const introInput  = includeIntro ? nextInput++ : -1;
  const outroInput  = includeOutro ? nextInput++ : -1;

  // Trim scenes from the single page.webm
  sceneClips.forEach((c, i) => {
    const startS = ((c.trimStartMs ?? 0) / 1000).toFixed(3);
    const endS   = ((c.trimEndMs   ?? 0) / 1000).toFixed(3);
    segments.push(`[0:v]trim=start=${startS}:end=${endS},setpts=PTS-STARTPTS[s${i}]`);
  });
  // Tag intro/outro inputs
  if (introInput >= 0) segments.push(`[${introInput}:v]setpts=PTS-STARTPTS[intro]`);
  if (outroInput >= 0) segments.push(`[${outroInput}:v]setpts=PTS-STARTPTS[outro]`);

  // Build the joined chain: [intro?] -> [s0..sN] -> [outro?]
  type Step = { label: string; durationMs: number };
  const order: Step[] = [];
  if (introInput >= 0) order.push({ label: "[intro]", durationMs: introResolved!.durationMs });
  sceneClips.forEach((c, i) => order.push({ label: `[s${i}]`, durationMs: c.durationMs }));
  if (outroInput >= 0) order.push({ label: "[outro]", durationMs: outroResolved!.durationMs });

  let prevLabel = order[0].label;
  let prevDuration = order[0].durationMs;
  for (let i = 1; i < order.length; i++) {
    // Transition into intro: none (intro is first)
    // Transition into a scene s_k: from per-scene config or default
    // Transition into outro: from frontmatter default
    const isSceneIn = order[i].label.startsWith("[s");
    const sceneIndex = isSceneIn ? Number(order[i].label.slice(2, -1)) : -1;
    const t: TransitionConfig = isSceneIn
      ? resolveTransition(opts.ast, sceneIndex)

```ts
type SubSegment =
  | { kind: "real";       startMs: number; endMs: number }
  | { kind: "ff";         startMs: number; endMs: number; factor: number }
  | { kind: "skip";       startMs: number; endMs: number };

function buildSubSegments(scene: ManifestSceneEntry, markers: ManifestMarker[]): SubSegment[] {
  const sceneMarkers = markers
    .filter(m => m.sceneIndex === scene.index)
    .sort((a, b) => a.tStartMs - b.tStartMs);
  const segs: SubSegment[] = [];
  let cursor = scene.tStartMs;
  for (const m of sceneMarkers) {
    if (m.tStartMs > cursor) segs.push({ kind: "real", startMs: cursor, endMs: m.tStartMs });
    if (m.kind === "fast_forward") segs.push({ kind: "ff", startMs: m.tStartMs, endMs: m.tEndMs, factor: m.factor });
    else                            segs.push({ kind: "skip", startMs: m.tStartMs, endMs: m.tEndMs });
    cursor = m.tEndMs;
  }
  if (cursor < scene.tEndMs) segs.push({ kind: "real", startMs: cursor, endMs: scene.tEndMs });
  return segs;
}

function emitSceneFilter(scene: ManifestSceneEntry, markers: ManifestMarker[], outLabel: string): { lines: string[]; durationMs: number } {
  const subs = buildSubSegments(scene, markers).filter(s => s.kind !== "skip");
  if (subs.length === 0) {
    return { lines: [`color=c=black:s=2x2:d=0.001${outLabel}`], durationMs: 0 };
  }
  const lines: string[] = [];
  const subLabels: string[] = [];
  let total = 0;
  subs.forEach((s, idx) => {
    const lbl = `[${outLabel.slice(1, -1)}_${idx}]`;
    const startS = (s.startMs / 1000).toFixed(3);
    const endS   = (s.endMs   / 1000).toFixed(3);
    if (s.kind === "real") {
      lines.push(`[0:v]trim=start=${startS}:end=${endS},setpts=PTS-STARTPTS${lbl}`);
      total += s.endMs - s.startMs;
    } else { // ff
      lines.push(`[0:v]trim=start=${startS}:end=${endS},setpts=(PTS-STARTPTS)/${s.factor}${lbl}`);
      total += (s.endMs - s.startMs) / s.factor;
    }
    subLabels.push(lbl);
  });
  if (subLabels.length === 1) {
    // Re-label single sub as scene output.
    lines[lines.length - 1] = lines[lines.length - 1].replace(subLabels[0], outLabel);
  } else {
    lines.push(`${subLabels.join("")}concat=n=${subLabels.length}:v=1:a=0${outLabel}`);
  }
  return { lines, durationMs: Math.round(total) };
}
```
